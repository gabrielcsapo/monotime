module.exports = (interval, amount) => {
    let self = {
        total: 0
    };

    switch (interval) {
        case 'hourly':

        case 'bi-weekly':
            self.total = amount * 26;

        case 'monthly':
            self.total = amount * 12;

    /**

    };
